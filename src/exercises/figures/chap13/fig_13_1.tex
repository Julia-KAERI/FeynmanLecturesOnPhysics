% SPACETIME DIAGRAM - EUCLIDEAN ROTATION
\begin{tikzpicture}[scale=1.7]
    \def\th{45};
    \def\ph{-30};
    \coordinate (Q) at (-2, 0);
    \coordinate (O) at (0, 0);
    \coordinate (P1) at ( {2*cos(\th)}, {2*sin(\th)});
    \coordinate (P2) at ( {2*cos(\ph)}, {2*sin(\ph)});

    \draw[-{Latex[length=3mm, width=2mm]},, very thick] (Q) -- (O) node[midway, above] {$\pi^0$};
    % 기본적인 물결 화살표
    %\draw[->, decorate, decoration={snake}] (O) -- (P1);
    
    % 물결의 진폭(amplitude)과 주기(segment length) 조절
    \draw[-{Latex[length=3mm, width=2mm]}, decorate, decoration={snake, amplitude=1mm, segment length=7.5mm}] (O) -- (P1);
    \draw[-{Latex[length=3mm, width=2mm]}, decorate, decoration={snake, amplitude=-1mm, segment length=7.5mm}] (O) -- (P2);
    
    \draw[dashed, very thick] (O) -- (2.5, 0);
    \draw [->] (1, 0) arc (0:{\th}:1cm) node[midway, right] {$\theta_1$};
    \draw [->] (1.2, 0) arc (0:{(\ph+2)}:1.2cm) node[midway, right] {$\theta_2$};

\end{tikzpicture}
